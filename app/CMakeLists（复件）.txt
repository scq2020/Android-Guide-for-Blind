# Sets the minimum version of CMake required to build the native library.  
cmake_minimum_required(VERSION 3.4.1)

# 打印当前的构建类型  
message(STATUS "Build Type: ${CMAKE_BUILD_TYPE}")

# 检查编译器是否支持 C++14  
include(CheckCXXCompilerFlag)  
CHECK_CXX_COMPILER_FLAG("-std=c++14" COMPILER_SUPPORTS_CXX14)  
CHECK_CXX_COMPILER_FLAG("-std=c++11" COMPILER_SUPPORTS_CXX11)  
CHECK_CXX_COMPILER_FLAG("-std=c++0x" COMPILER_SUPPORTS_CXX0X)  

# 设置 C++ 标准  
if(COMPILER_SUPPORTS_CXX14)  
    set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -std=c++14")  
    message(STATUS "Using C++14 flag")  
elseif(COMPILER_SUPPORTS_CXX11)  
    set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -std=c++11")  
    message(STATUS "Using C++11 flag")  
elseif(COMPILER_SUPPORTS_CXX0X)  
    set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -std=c++0x")  
    message(STATUS "Using C++0x flag")  
else()  
    message(FATAL_ERROR "The compiler ${CMAKE_CXX_COMPILER} has no C++14 or C++11 support. Please use a different C++ compiler.")  
endif()  

# 现代 CMake 方式设置 C++ 标准  
set(CMAKE_CXX_STANDARD 14)  
set(CMAKE_CXX_STANDARD_REQUIRED ON)  
set(CMAKE_CXX_EXTENSIONS OFF)  

# 检查 OpenMP  
find_package(OpenMP)  
# 完全禁用 OpenMP
add_definitions(-DNO_OPENMP)
set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -frtti")
set(CMAKE_C_FLAGS "${CMAKE_C_FLAGS} -frtti")
message(STATUS "Building without OpenMP support")

# 检查 libomp.so 是否存在于 jniLibs 目录中  
if(EXISTS "${CMAKE_SOURCE_DIR}/src/main/jniLibs/${ANDROID_ABI}/libomp.so")  
    message(STATUS "Found libomp.so in jniLibs/${ANDROID_ABI}")  
else()  
    message(WARNING "libomp.so not found in jniLibs/${ANDROID_ABI}, you may need to add it")  
endif()  

# 调试信息  
message(STATUS "C++ Flags: ${CMAKE_CXX_FLAGS}")  
message(STATUS "C++ Standard: ${CMAKE_CXX_STANDARD}")
